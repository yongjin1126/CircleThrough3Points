#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> board, int k) {
      long long sum = 0;
    int R = (int)board.size();
    for (int i = 0; i < R; ++i) {
        int C = (int)board[i].size();
        // jëŠ” 0..min(C-1, k-i)
        int jmax = min(C - 1, k - i);
        for (int j = 0; j <= jmax; ++j) if (i + j <= k) sum += board[i][j];
    }
    return (int)sum;
}
